import { Component, Input } from '@angular/core';

type Variants = 'primary' | 'secondary' | 'success' | 'danger' | 'warning' | 'info' | 'light' | 'dark' | 'link';

const classes = {
  'primary': 'bg-blue-500 hover:bg-blue-700 text-white py-2 px-4 rounded',
  'secondary': 'bg-gray-500 hover:bg-gray-700 text-white py-2 px-4 rounded',
  'success': 'bg-green-500 hover:bg-green-700 text-white py-2 px-4 rounded',
  'danger': 'bg-red-500 hover:bg-red-700 text-white py-2 px-4 rounded',
  'warning': 'bg-yellow-500 hover:bg-yellow-700 text-white py-2 px-4 rounded',
  'info': 'bg-teal-500 hover:bg-teal-700 text-white py-2 px-4 rounded',
  'light': 'bg-gray-200 hover:bg-gray-400 text-gray-800 py-2 px-4 rounded',
  'dark': 'bg-gray-700 hover:bg-gray-900 text-white py-2 px-4 rounded',
  'link': 'bg-transparent hover:bg-blue-500 text-blue-700 font-semibold hover:text-white py-2 px-4 border border-blue-500 hover:border-transparent rounded',
};

@Component({
  selector: 'divi-button',
  templateUrl: './button.component.html',
  styleUrls: ['./button.component.scss']
})
export class ButtonComponent {
  @Input() type: 'button' | 'menu' | 'reset' | 'submit' = 'button';
  @Input() variant: Variants = 'primary';
  applyClassVariants = (variant: Variants): string => classes[variant];
}

<button [type]="type" [ngClass]="applyClassVariants(variant)">
  <ng-content></ng-content>
</button>
